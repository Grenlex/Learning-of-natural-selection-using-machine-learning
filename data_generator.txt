#!/bin/bash
echo "Generation started"
mkdir locus_$1
for ((i=0; i < 20; i++))
do
real_part=$(($RANDOM%9+1))
final_num_1="0."$real_part
real_part=$(($RANDOM%9+1))
final_num_2="0."$real_part
real_part=$(($RANDOM%9+1))
final_num_3="0."$real_part
echo "S	F	0	"$1"	"$final_num_1"	"$final_num_2"	"$final_num_3 > selection.txt
SELAM -d demography.txt -o output.txt -s selection.txt -h
SELAM_STATS -i population_output.txt -a 0.05 > locus_$1/"selection_"$2"_frequency_"$i".txt"
rm population_output.txt
done
echo "Generation finished successfully"
